  <script>
    // This value links dynamically to your Function URL once deployed to Azure
    const AZURE_API_BASE_URL = window.location.origin.includes("localhost") 
      ? "http://localhost:7071/api" 
      : window.location.origin.replace("-ui", "-api") + "/api"; 

    function showView(viewId, element) {
      document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
      document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
      document.getElementById('view-' + viewId).classList.add('active');
      if(element) element.classList.add('active');
    }

    async function triggerAzureSync() {
      st("Syncing Teams Sheets...");
      try {
        const res = await fetch(`${AZURE_API_BASE_URL}/syncDemandData`, { method: 'POST' });
        alert(await res.text());
      } catch(e) { alert("Sync connection failed."); }
      ready();
    }

    async function fetchEarliestActiveWeek() {
      const model = document.getElementById('filterModel').value;
      try {
        const res = await fetch(`${AZURE_API_BASE_URL}/getEarliestWeek`, { method: 'POST', body: JSON.stringify({ model }) });
        if(res.ok) document.getElementById('filterWeek').value = await res.text();
      } catch(e) { console.error(e); }
    }

    async function runLabelExport() {
      const model = document.getElementById('filterModel').value;
      const weekNum = document.getElementById('filterWeek').value;
      st("Processing label export matrices...");
      try {
        const res = await fetch(`${AZURE_API_BASE_URL}/getFilteredCarpetExport`, { method: 'POST', body: JSON.stringify({ model, weekNum }) });
        const data = await res.json();
        alert(`Export processing complete. Rows prepared: ${data.count}`);
      } catch(e) { alert("Failed to fetch label compilation matrix."); }
      ready();
    }

    async function submitShipping(type) {
      const chassisList = document.getElementById('chassisInput').value;
      st("Logging outbound shipment records...");
      try {
        const res = await fetch(`${AZURE_API_BASE_URL}/processShipping`, { method: 'POST', body: JSON.stringify({ chassisList, type }) });
        const data = await res.json();
        alert(data.message);
        document.getElementById('chassisInput').value = "";
      } catch(e) { alert("Outbound logistics update failed."); }
      ready();
    }

    async function fetchLiveExplorerData() {
      const tabName = document.getElementById('tabSelect').value;
      st("Fetching historical table arrays...");
      try {
        const res = await fetch(`${AZURE_API_BASE_URL}/getTabData`, { method: 'POST', body: JSON.stringify({ tabName }) });
        const rows = await res.json();
        
        const headTr = document.getElementById('tableHeader'); headTr.innerHTML = "";
        const bodyT = document.getElementById('tableBody'); bodyT.innerHTML = "";
        
        if(rows.length > 0) {
          rows[0].forEach(h => headTr.innerHTML += `<th>${h}</th>`);
          rows.slice(1).forEach(r => {
            let trHtml = "<tr>"; r.forEach(cell => trHtml += `<td>${cell}</td>`); trHtml += "</tr>";
            bodyT.innerHTML += trHtml;
          });
        }
      } catch(e) { alert("Error reading table array datasets."); }
      ready();
    }

    function st(text) { document.getElementById('status').innerHTML = `<i class="fas fa-spinner fa-spin"></i> ${text}`; }
    function ready() { document.getElementById('status').innerHTML = `<i class="fas fa-circle-check"></i> Connected to Azure API`; }
    window.onload = fetchEarliestActiveWeek;
  </script>
