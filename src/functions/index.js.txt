const { app } = require('@azure/functions');
const { Client } = require("@microsoft/microsoft-graph-client");
const { ClientSecretCredential } = require("@azure/identity");
const { TokenCredentialAuthenticationProvider } = require("@microsoft/microsoft-graph-client/authProviders/azureTokenCredentials");

// Fetch Environment Variables from Azure
const tenantId = process.env.AZURE_TENANT_ID;
const clientId = process.env.AZURE_CLIENT_ID;
const clientSecret = process.env.AZURE_CLIENT_SECRET;
const SITE_ID = process.env.TEAMS_SHAREPOINT_SITE_ID;
const MASTER_DRIVE_ITEM_ID = process.env.MASTER_XLSX_FILE_ID;
const DAILY_DRIVE_ITEM_ID = process.env.DAILY_XLSX_FILE_ID;

// Excel Sheet to Generic Generated Table Mapping Configuration
const TABLE_MAPPING = {
    "AM3": "Table1", "AM5": "Table2", "AM6": "Table3", "AM003": "Table4", "AM8": "Table5",
    "AP TLS AM3 BOOTKITS": "Table6", "AP TLS AM7": "Table7", "Bootkit Shipped": "Table8",
    "Carpet Shipped": "Table9", "Bootkit Master Data": "Table10", "Carpet Masterdata": "Table11",
    "VCS Master": "Table12"
};

const credential = new ClientSecretCredential(tenantId, clientId, clientSecret);
const authProvider = new TokenCredentialAuthenticationProvider(credential, { scopes: ["https://graph.microsoft.com/.default"] });
const graphClient = Client.initWithMiddleware({ authProvider });

function parseBody(bodyString) { try { return JSON.parse(bodyString); } catch(e) { return {}; } }

/** 1. FUNCTION: SYNC DEMAND DATA (ROW 2 DOWNWARD) **/
app.http('syncDemandData', {
    methods: ['POST'], authLevel: 'anonymous',
    handler: async (request, context) => {
        context.log('Sync processing triggered.');
        const coreSheets = ["AM3", "AM5", "AM6", "AM003", "AM8"];
        
        for (const sheetName of coreSheets) {
            try {
                const targetTableName = TABLE_MAPPING[sheetName];
                if (!targetTableName) continue;

                // Pull values from Source Daily spreadsheet table inside Teams
                const sourceUrl = `/sites/${SITE_ID}/drive/items/${DAILY_DRIVE_ITEM_ID}/workbook/worksheets/${sheetName}/tables/${targetTableName}/range`;
                const sourceRange = await graphClient.api(sourceUrl).get();
                if (!sourceRange.values || sourceRange.values.length <= 1) continue;

                // Slice data to pull strictly from Row 2 downwards
                const sourceDataOnly = sourceRange.values.slice(1);

                // Clear old rows inside Target Master file table while leaving Row 1 headers intact
                const targetTableUrl = `/sites/${SITE_ID}/drive/items/${MASTER_DRIVE_ITEM_ID}/workbook/worksheets/${sheetName}/tables/${targetTableName}`;
                const currentRows = await graphClient.api(`${targetTableUrl}/rows`).get();
                if (currentRows.value && currentRows.value.length > 0) {
                    for (let i = currentRows.value.length - 1; i >= 0; i--) {
                        await graphClient.api(`${targetTableUrl}/rows/$${i}`).delete();
                    }
                }
                
                // Post data immediately under row 1 headers
                await graphClient.api(`${targetTableUrl}/rows`).post({ values: sourceDataOnly });
            } catch (err) {
                context.log(`Failed core sheet sync: ${sheetName}`, err);
            }
        }
        return { status: 200, body: "Sync Complete." };
    }
});

/** 2. FUNCTION: FILTERED CARPET EXPORT **/
app.http('getFilteredCarpetExport', {
    methods: ['POST'], authLevel: 'anonymous',
    handler: async (request, context) => {
        const reqData = parseBody(await request.text());
        const { model, weekNum } = reqData;
        try {
            let sheetsToProcess = (model === "AM5") ? ["AP TLS AM5", "AP TLS AM7"] : ["AP TLS " + model];
            let combinedFilteredData = [];
            let labelChassis = [["Chassis Number"]];

            for (const sheetName of sheetsToProcess) {
                const tTable = TABLE_MAPPING[sheetName];
                const dataRange = await graphClient.api(`/sites/${SITE_ID}/drive/items/${MASTER_DRIVE_ITEM_ID}/workbook/worksheets/${sheetName}/tables/${tTable}/range`).get();
                if (!dataRange.values || dataRange.values.length === 0) continue;
                const data = dataRange.values;
                if (combinedFilteredData.length === 0) combinedFilteredData.push(data[0]);

                for (let i = 1; i < data.length; i++) {
                    const row = data[i];
                    if (row[2]?.toString() === weekNum?.toString() && row[6] === 0 && row[1] === 0) {
                        combinedFilteredData.push(row);
                        if (row[5]) labelChassis.push([row[5]]);
                    }
                }
            }
            return { status: 200, headers: { "Content-Type": "application/json" }, body: JSON.stringify({ count: combinedFilteredData.length - 1, labelData: labelChassis, rawExport: combinedFilteredData }) };
        } catch (e) {
            return { status: 500, body: "Export Failed: " + e.message };
        }
    }
});

/** 3. FUNCTION: PROCESS SHIPPING LOGS **/
app.http('processShipping', {
    methods: ['POST'], authLevel: 'anonymous',
    handler: async (request, context) => {
        const reqData = parseBody(await request.text());
        const { chassisList, type } = reqData;
        try {
            const sheetName = (type === 'bootkit') ? "Bootkit Shipped" : "Carpet Shipped";
            const targetTable = TABLE_MAPPING[sheetName];
            const shipDate = new Date().toISOString().replace('T', ' ').substring(0, 16);
            const cleanList = chassisList.split(/[\n,]+/).map(item => item.trim()).filter(item => item !== "");
            
            if (cleanList.length === 0) return { status: 400, body: "Error: No chassis numbers provided." };
            const rowsToAdd = cleanList.map(chassis => [chassis, shipDate]);

            await graphClient.api(`/sites/${SITE_ID}/drive/items/${MASTER_DRIVE_ITEM_ID}/workbook/worksheets/${sheetName}/tables/${targetTable}/rows`).post({ values: rowsToAdd });
            return { status: 200, headers: { "Content-Type": "application/json" }, body: JSON.stringify({ message: `Success: ${cleanList.length} logged.` }) };
        } catch (e) {
            return { status: 500, body: "Error processing shipping: " + e.message };
        }
    }
});

/** 4. FUNCTION: LIVE DATA EXPLORER READS **/
app.http('getTabData', {
    methods: ['POST'], authLevel: 'anonymous',
    handler: async (request, context) => {
        const reqData = parseBody(await request.text());
        const { tabName } = reqData;
        try {
            const targetTable = TABLE_MAPPING[tabName.trim()];
            const res = await graphClient.api(`/sites/${SITE_ID}/drive/items/${MASTER_DRIVE_ITEM_ID}/workbook/worksheets/${tabName.trim()}/tables/${targetTable}/range`).get();
            return { status: 200, headers: { "Content-Type": "application/json" }, body: JSON.stringify(res.values || []) };
        } catch (e) {
            return { status: 200, headers: { "Content-Type": "application/json" }, body: JSON.stringify([["System Access Error: " + e.message]]) };
        }
    }
});

/** 5. FUNCTION: EARLIEST ACTIVE WEEK FINDER **/
app.http('getEarliestWeek', {
    methods: ['POST'], authLevel: 'anonymous',
    handler: async (request, context) => {
        const reqData = parseBody(await request.text());
        const { model } = reqData;
        try {
            let sheetsToProcess = (model === "AM5") ? ["AP TLS AM5", "AP TLS AM7"] : ["AP TLS " + model];
            let weeks = [];
            for (const name of sheetsToProcess) {
                const tTable = TABLE_MAPPING[name];
                const res = await graphClient.api(`/sites/${SITE_ID}/drive/items/${MASTER_DRIVE_ITEM_ID}/workbook/worksheets/${name}/tables/${tTable}/range`).get();
                if (!res.values) continue;
                for (let i = 1; i < res.values.length; i++) {
                    const r = res.values[i];
                    const weekVal = parseInt(r[2]);
                    if (!isNaN(weekVal) && r[6] === 0 && r[1] === 0) weeks.push(weekVal);
                }
            }
            return { status: 200, body: weeks.length > 0 ? Math.min(...weeks).toString() : "" };
        } catch (e) {
            return { status: 200, body: "" };
        }
    }
});
